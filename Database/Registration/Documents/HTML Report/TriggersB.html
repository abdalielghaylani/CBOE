REGDB.PARAMETER_ID_TRIG


CREATE TRIGGER "REGDB"."PARAMETER_ID_TRIG" BEFORE INSERT
 	ON "REGDB"."PARAMETERS" REFERENCING NEW AS NEW OLD AS OLD FOR EACH ROW
 BEGIN
 SELECT SEQ_PARAMETERS.NEXTVAL INTO :NEW.PARAMETER_ID  FROM DUAL;
 END;
 /
