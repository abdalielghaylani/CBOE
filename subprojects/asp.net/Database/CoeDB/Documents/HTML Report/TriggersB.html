COEDB.PEOPLE_TRIG


CREATE TRIGGER "COEDB"."PEOPLE_TRIG" BEFORE INSERT
 	ON "COEDB"."PEOPLE" REFERENCING NEW AS NEW OLD AS OLD FOR EACH ROW
 BEGIN
    SELECT people_seq.NEXTVAL
      INTO :NEW.person_id
      FROM DUAL;
 END;
 /

 	ON "COEDB"."CS_SECURITY_PRIVILEGES" REFERENCING NEW AS NEW OLD AS OLD FOR EACH ROW

                 || TO_CHAR (:OLD.css_edit_user) || '|'

                 || TO_CHAR (:OLD.css_create_role) || '|'
