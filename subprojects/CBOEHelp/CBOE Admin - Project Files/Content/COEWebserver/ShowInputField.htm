Subroutine

ShowInputField 
      dbkey,formgroup,table.field,fieldtype,width

Description

Shows input box

Parameters

dbkey = <dataview name>  

formgroup = <formgroup name>

table.field = <table and field name using dot syntax>  

fieldtype = 0|1|2|3|5|7|8|21
         0 : text; no validation
         1 : integer validation
         2 : long validation
         3 : positive integer validation
         5 : Molecular Weight (positive float validation)
         7 : memo; no validation
         8 : date; validation dd/mm/yyyy
         21 : CAS number; validation

width = <width of text box to display, or rows of textarea>

Example

ShowInputField 
      dbkey,formgroup,"Synonyms_r.Synonym_r",0,"30"
